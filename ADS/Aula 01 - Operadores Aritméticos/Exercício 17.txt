public class AreaTriangulo {

    public static void main(String[] args) {

        // Definindo a base e a altura do triângulo
        double base = 10.0;   // você pode alterar este valor
        double altura = 5.0;  // você pode alterar este valor

        // Calculando a área do triângulo
        double area = (base * altura) / 2;

        // Exibindo o resultado
        System.out.printf("A área do triângulo com base %.2f e altura %.2f é: %.2f%n", base, altura, area);
    }
}
